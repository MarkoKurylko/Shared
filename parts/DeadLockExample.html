public class DeadLock extends Thread {

    Thread other;

    @Override
    public void run() {
        try {
            other.join();
        } catch (InterruptedException ex) {
        }
    }

    public static void main(String[] args) {
        DeadLock t1 = new DeadLock();
        DeadLock t2 = new DeadLock();

        t1.other = t2;
        t2.other = t1;

        t1.start();
        t2.start();
    }
}

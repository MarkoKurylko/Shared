@Getter
public class ClassWithBuilder {
    private final String strProperty;
    private final int intProperty;

    ClassWithBuilder(Builder builder) {
        this.strProperty = builder.strProperty;
        this.intProperty = builder.intProperty;
    }

    public static class Builder {

        private String strProperty;
        private int intProperty;

        public Builder withStrProperty(String strProperty) {
            this.strProperty = strProperty;
            return this;
        }

        public Builder withIntProperty(int intProperty) {
            this.intProperty = intProperty;
            return this;
        }

        public ClassWithBuilder build() {
            return new ClassWithBuilder(this);
        }
    }

    public static void main(String[] args) {
        ClassWithBuilder cwb = new ClassWithBuilder.Builder()
                .withStrProperty("Java Builder Pattern")
                .withIntProperty(100)
                .build();
    }
}

class Liskov {

    public static void main(String[] args) {
        test(new ExtOK());
        test(new ExtFail()); // Compile error, Java do force LSP
    }

    static void test(Liskov.Base base) {
        base.method();
    }

    static class Base {
        public void method() {
            System.out.println("FooBar");
        }
    }

    // This class is OK
    static class ExtOK extends Base {
        public void method() {
            System.out.println("Ext");
        }
    }

    // This class is violating Liskov substitution
    static class ExtFail extends Base {
        @Override
        private void method() {
            System.out.println("ExtFail");
        }
    }

}
